// ── Counter animation ──────────────────────────────────────────
function animateCount(el, target, duration, suffix = '') {
  const start = performance.now();
  const update = (now) => {
    const progress = Math.min((now - start) / duration, 1);
    const ease = 1 - Math.pow(1 - progress, 3);
    el.innerHTML = Math.floor(ease * target).toLocaleString('fr-FR') + suffix;
    if (progress < 1) requestAnimationFrame(update);
  };
  requestAnimationFrame(update);
}

// ── Sparklines ──────────────────────────────────────────────────
function buildSparkline(containerId, data, activeIdx) {
  const el = document.getElementById(containerId);
  if (!el) return;
  const max = Math.max(...data);
  data.forEach((v, i) => {
    const bar = document.createElement('div');
    bar.className = 'spark-bar' + (i === activeIdx ? ' active' : '');
    bar.style.height = '0px';
    el.appendChild(bar);
    setTimeout(() => {
      bar.style.transition = `height .6s ${i * 0.06}s cubic-bezier(.4,0,.2,1)`;
      bar.style.height = Math.max(4, (v / max) * 28) + 'px';
    }, 300);
  });
}

// ── Revenue chart ───────────────────────────────────────────────
function buildRevenueChart() {
  const data    = [32000, 41500, 28000, 55000, 48000, 61000, 52000];
  const labels  = ['L', 'M', 'M', 'J', 'V', 'S', 'D'];
  const chartEl = document.getElementById('chart-revenue');
  const labEl   = document.getElementById('chart-labels');
  if (!chartEl) return;
  const max = Math.max(...data);
  data.forEach((v, i) => {
    const bar = document.createElement('div');
    bar.style.cssText = `flex:1; border-radius:4px 4px 0 0; background:${i===5?'var(--accent)':'rgba(0,212,184,0.2)'}; height:0; transition: height .8s ${i*.08}s cubic-bezier(.4,0,.2,1);`;
    chartEl.appendChild(bar);
    setTimeout(() => { bar.style.height = Math.max(4, (v/max)*36)+'px'; }, 400);
  });
  labels.forEach(l => {
    const s = document.createElement('span'); s.textContent = l;
    labEl.appendChild(s);
  });
}

// ── Funnel bars ─────────────────────────────────────────────────
function animateFunnels() {
  document.querySelectorAll('.funnel-bar-fill').forEach((el, i) => {
    const pct = parseInt(el.dataset.pct);
    setTimeout(() => { el.style.width = pct + '%'; }, 400 + i * 120);
  });
}
